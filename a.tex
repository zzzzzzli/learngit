你好，学习 Git 分布式版本控制系统。\par
\begin{enumerate}
  \item
  \begin{itemize}
    \item \verb|git init|, 初始化目录为一个 Git 仓库
    \item \verb|git add|, 添加文件到 Git, 准备提交到仓库
    \item \verb|git commit -m "wirte somthing about your changes"|, 提交文件到仓库
  \end{itemize}
  \item 修改你的文件
  \begin{itemize}
    \item \verb|git status| 告诉你哪个文件被修改过
    \item \verb|git diff|
  \end{itemize}
\end{enumerate}
